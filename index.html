// 預設日期 = 今天

(function setToday(){

  const pad = n => String(n).padStart(2,'0')

  const d = new Date()

  document.getElementById('a').value = `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`

})()


// 將資料送到 Google Apps Script（把 YOUR_GAS_URL 換成你的部署網址）

const GAS_URL = "https://script.google.com/macros/s/AKfycbwzMAgcqCzIEJ-gsprh57mRt9QAHtf-jEvHI_4hvefBMmzR0SrydRZIlCQNiwJiMo8/exec";


document.getElementById('f').addEventListener('submit', async (evt)=>{

  evt.preventDefault();

  const picked = document.querySelector('input[name="etype"]:checked')

  const eField = document.getElementById('e')

  const eValue = eField.value.trim()

  if(!picked){ alert('請先選擇「事件類型」。'); return }

  if(!eValue){ eField.classList.add('invalid'); eField.focus(); alert('「事件」為必填，請輸入內容。'); return }


  const payload = {

    "事件類型": picked.value,

    "日期": document.getElementById('a').value,

    "單位": document.getElementById('b').value || "",

    "客戶": document.getElementById('c').value || "",

    "品項": document.getElementById('d').value || "",

    "事件": eValue

  }


  try{

    const res = await fetch(GAS_URL, {

      method: 'POST',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify(payload)

    })

    const data = await res.json()

    if(data.ok){

      alert('已送出！')

      // 可選：送出後清空

      // document.getElementById('f').reset()

      // 或者：回到上一頁 / 轉跳

      // location.href = 'thanks.html'

    }else{

      alert('送出失敗：' + (data.error || '未知錯誤'))

    }

  }catch(err){

    alert('連線失敗，請稍後再試：' + err)

  }

})
